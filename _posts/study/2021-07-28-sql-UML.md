# INSERT
- 데이터 입력
- INSERT INTO 테이블명 (열) VALUES (값);

# UPDATE
- 정보 수정
- UPDATE 테이블명 SET 칼럼명 = 수정 값;

# DELETE
- 데이터 삭제
- DELETE FROM 테이블명;

# SELECT
- 데이터  조회
- SELECT * FROM 테이블명; (전체 조회)
- SELECT 보고싶은 칼럼 선택 가능 FROM 테이블명;
