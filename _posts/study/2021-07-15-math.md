# Math.pow(x,y) -> x의 y승, 제곱함수
- Math.pow(7, 2);    // 49

# Math.round(x) -> 반올림
- Math.round(10.6);    // 11

# Math.ceil(x) -> 올림
- Math.ceil(10.2);    // 11

# Math.floor(x) -> 내림
- Math.floor(10.2);    // 10

# Math.sqrt(x) -> 제곱근 구하는 함수
- Math.sqrt(9);    // 3 

# Math.random() -> 랜덤출력
- Math.random();    // 0.6709366008510558 등 과 같이 랜덤 소수점 출력 
- Math.round(100*Math.random()); // 100이하의 자연수 랜덤출력
